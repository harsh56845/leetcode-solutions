SELECT
    p.product_id,
    coalesce((ROUND(SUM(p.price*us.units)/SUM(us.units),2)),0) as average_price
FROM Prices p
-- WHERE us.purchase_date>=p.start_date and us.purchase_date<=p.end_date
LEFT JOIN UnitsSold us ON p.product_id=us.product_id
AND us.purchase_date>=p.start_date and us.purchase_date<=p.end_date
GROUP BY p.product_id
